47 static Eigen::SparseMatrix<EleType> second_order_central_difference(
48 size_t n) {
49 // Grid spacing
50 EleType h = grid_spacing_h(n);
51
52 // Unfilled difference matrix
53 Eigen::SparseMatrix<EleType> D(n, n);
54
55 // Assemble the difference matrix by direct insertion of values
56 size_t n_diagonals = 3;
57 D.reserve(Eigen::VectorXi::Constant(n, n_diagonals));
58 for (int i = 0; i < n; ++i) {
59 D.insert(i, i) = -2.0;
60 if (i > 0) { // handle first row
61 D.insert(i, i - 1) = 1.0;
62 }
63 if (i < n - 1) { // handle last row
64 D.insert(i, i + 1) = 1.0;
65 }
66 }
67
68 // Finite differences requires this division
69 D = D / (h * h);
70
71 return D;
72 }
