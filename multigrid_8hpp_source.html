152 Multigrid(AMG::InterpolatorBase<EleType>* interpolator_,

177 std::to_string(A.rows()) + " and " + std::to_string(b.rows());
178 throw(std::invalid_argument(errormsg));

181 // Initialize linear system info
182 level_to_coefficient_matrix.resize(n_levels);
183 level_to_soln.resize(n_levels);
184 level_to_rhs.resize(n_levels);
185 level_to_residual.resize(n_levels);
186 level_to_n_dofs.resize(n_levels);
187
188 // Initialize index for coarsest grid
189 coarsest_grid_ix = n_levels - 1;
190
191 // Initialize the finest coefficients matrix
192 size_t n_fine_dofs = A.rows();
193 level_to_n_dofs[finest_grid_ix] = n_fine_dofs;
194 level_to_coefficient_matrix[finest_grid_ix] = A;
195
196 // Initialize the finest solutions vector
197 Eigen::Matrix<EleType, -1, 1> u(n_fine_dofs);
198 u.setZero();
199 level_to_soln[finest_grid_ix] = u;
200
201 // Initialize the finest right hand side
202 level_to_rhs[finest_grid_ix] = b;
203
204 // Initialize the finest residual
205 level_to_residual[finest_grid_ix] = b - A * u;
206
207 // Use the finest level grid info to construct the linear interpolators
208 // needed to construct the remaining grid stuff
209 // TODO: could do while coarsest is greater than max coarse ndofs
210 size_t n_H_dofs;
211 size_t n_h_dofs;
212 for (size_t level = 1; level < n_levels; ++level) {
213 // Make restriction/prolongation matrices
214 n_h_dofs = level_to_n_dofs[level - 1];
215 n_H_dofs = n_H_dofs_from_n_h_dofs(n_h_dofs);
216 level_to_n_dofs[level] = n_H_dofs;
217 interpolator->make_operators(n_h_dofs, n_H_dofs, level - 1);
218
219 // Make coefficient matrix using interpolation operators
220 auto R_h = interpolator->get_R(level - 1);
221 auto A_h = level_to_coefficient_matrix[level - 1]; // finer matrix
222 auto P_h = interpolator->get_P(level - 1);
223 auto A_H = R_h * (A_h * P_h); // coarser matrix
224 level_to_coefficient_matrix[level] = A_H;
225
226 // Make solution vector
227 Eigen::Matrix<EleType, -1, 1> u_H(n_H_dofs);
228 u_H.setZero();
229 level_to_soln[level] = u_H;
230
231 // Make right hand side
232 Eigen::Matrix<EleType, -1, 1> rhs_H(n_H_dofs);
233 rhs_H.setZero();
234 level_to_rhs[level] = rhs_H;
235
236 // Make residual vector
237 level_to_residual[level] = rhs_H - A_H * u_H;
238 }

240 // Initialize the coarse solver

242 level_to_coefficient_matrix[coarsest_grid_ix]);
243 coarse_direct_solver.factorize(
244 level_to_coefficient_matrix[coarsest_grid_ix]);

264 void vcycle() {
265 // At each level of the grid hiearchy, finest-to-coarsest:
266 for (size_t level = 0; level < n_levels; ++level) {
267 // 1. Apply a couple of smoothing iterations (pre-relaxation) to the
268 // current solution ui=Si(Ai,fi,ui)
269 smoother->smooth(level_to_coefficient_matrix[level], level_to_soln[level],
270 level_to_rhs[level]);
271
272 // 2. Find residual ei=fi−Aiui and restrict it to the coarser level
273 level_to_residual[level] =
274 level_to_rhs[level] -
275 level_to_coefficient_matrix[level] * level_to_soln[level];
276
277 if (level + 1 != n_levels) {
278 // initialize coarse solution to zero here refs [2-4].
279 level_to_soln[level + 1].setZero();
280
281 // f_{i+1} = R_{i} e_{i}
282 level_to_rhs[level + 1] =
283 interpolator->restriction(level_to_residual[level], level);
284 }
285 }
286
287 // Solve the coarsest system directly: uL=A−1LfL
288 level_to_soln[coarsest_grid_ix] =
289 coarse_direct_solver.solve(level_to_rhs[coarsest_grid_ix]);
290
291 // At each level of the grid hierarchy, coarsest-to-finest:
292 for (int level = coarsest_grid_ix - 1; level >= 0; --level) {
293 // 1. Update the current solution with the interpolated solution from the
294 // coarser level: ui=ui+Piui+1
295 level_to_soln[level] =
296 level_to_soln[level] +
297 interpolator->prolongation(level_to_soln[level + 1], level);
298
299 // 2. Apply a couple of smoothing iterations (post-relaxation) to the
300 // updated solution: ui=Si(Ai,fi,ui)
301 smoother->smooth(level_to_coefficient_matrix[level], level_to_soln[level],
302 level_to_rhs[level]);
303 }
304
305 return;
306 }

312 const Eigen::Matrix<EleType, -1, 1>& solve() {
313 size_t iter = 0;
314 EleType error = 100;
315 auto A = get_coefficient_matrix(finest_grid_ix);
316 auto b = get_rhs(finest_grid_ix);
317 while (iter < n_iters && error > tolerance) {
318 vcycle();
319 iter += 1;
320 if ((iter % compute_error_every_n_iters) == 0) {
321 auto u = get_soln(finest_grid_ix);
322 error = rss(A, u, b);
323
324 if (display_error) {
325 std::cout << "Iter: " << iter << " | Error: " << error << std::endl;
326 }
327 }
328 }
329
330 if (error <= tolerance)
331 std::cout << "AMG converged after " << iter << " iterations."
332 << std::endl;
333 else
334 std::cout << "AMG did not converge after " << iter << " iterations."
335 << std::endl;
336
337 return level_to_soln[finest_grid_ix];
338 }
