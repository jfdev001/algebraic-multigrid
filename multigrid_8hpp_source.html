197 // needed to construct the remaining grid stuff

199 size_t n_H_dofs;

204 n_H_dofs = n_H_dofs_from_n_h_dofs(n_h_dofs);

211 auto P_h = interpolator->get_P(level - 1);

253 void vcycle() {
254 // At each level of the grid hiearchy, finest-to-coarsest:
255 for (size_t level = 0; level < n_levels; ++level) {
256 // 1. Apply a couple of smoothing iterations (pre-relaxation) to the
257 // current solution ui=Si(Ai,fi,ui)
258 smoother->smooth(level_to_coefficient_matrix[level], level_to_soln[level],
259 level_to_rhs[level]);
260
261 // 2. Find residual ei=fi−Aiui and restrict it to the coarser level
262 level_to_residual[level] =
263 level_to_rhs[level] -
264 level_to_coefficient_matrix[level] * level_to_soln[level];
265
266 if (level + 1 != n_levels) {
267 // initialize coarse solution to zero here refs [2-4].
268 level_to_soln[level + 1].setZero();
269
270 // f_{i+1} = R_{i} e_{i}
271 level_to_rhs[level + 1] =
272 interpolator->restriction(level_to_residual[level], level);
273 }
274 }
275
276 // Solve the coarsest system directly: uL=A−1LfL
277 level_to_soln[coarsest_grid_ix] =
278 coarse_direct_solver.solve(level_to_rhs[coarsest_grid_ix]);
279
280 // At each level of the grid hierarchy, coarsest-to-finest:
281 for (int level = coarsest_grid_ix - 1; level >= 0; --level) {
282 // 1. Update the current solution with the interpolated solution from the
283 // coarser level: ui=ui+Piui+1
284 level_to_soln[level] =
285 level_to_soln[level] +
286 interpolator->prolongation(level_to_soln[level + 1], level);
287
288 // 2. Apply a couple of smoothing iterations (post-relaxation) to the
289 // updated solution: ui=Si(Ai,fi,ui)
290 smoother->smooth(level_to_coefficient_matrix[level], level_to_soln[level],
291 level_to_rhs[level]);
292 }
293
294 return;
295 }

301 const Eigen::Matrix<EleType, -1, 1>& solve() {
302 size_t iter = 0;
303 EleType error = 100;
304 auto A = get_coefficient_matrix(finest_grid_ix);
305 auto b = get_rhs(finest_grid_ix);
306 while (iter < n_iters && error > tolerance) {
307 vcycle();
308 iter += 1;
309 if ((iter % compute_error_every_n_iters) == 0) {
310 auto u = get_soln(finest_grid_ix);
311 error = rss(A, u, b);
312
313 if (display_error) {
314 std::cout << "Iter: " << iter << " | Error: " << error << std::endl;
315 }
316 }
317 }
318
319 if (error <= tolerance)
320 std::cout << "AMG converged after " << iter << " iterations."
321 << std::endl;
322 else
323 std::cout << "AMG did not converge after " << iter << " iterations."
324 << std::endl;

327 }
