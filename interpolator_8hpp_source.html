100 // Create prolongation matrix
101 size_t nnz = n_H_dofs * n_elements_per_columns;
102 Eigen::SparseMatrix<EleType> P(n_h_dofs, n_H_dofs);
103 P.reserve(nnz);
104
105 // Populate nonzeros in matrix and bounds check the rows
106 // TODO: Do these bound checks disrupt algorithm correctness?
107 std::vector<Eigen::Triplet<EleType>> P_coefficients;
108 P_coefficients.reserve(nnz);
109 size_t i = 0;
110 for (size_t j = 0; j < n_H_dofs; ++j) {
111 if (i < n_h_dofs)
112 P_coefficients.push_back(Eigen::Triplet<EleType>(i, j, 0.5));
113
114 if (i + 1 < n_h_dofs)
115 P_coefficients.push_back(Eigen::Triplet<EleType>(i + 1, j, 1.0));
116
117 if (i + 2 < n_h_dofs)
118 P_coefficients.push_back(Eigen::Triplet<EleType>(i + 2, j, 0.5));
119
120 i += n_elements_per_columns - 1;
121 }
122 P.setFromTriplets(P_coefficients.begin(), P_coefficients.end());
123
124 // Restriction matrix follows from prolongation matrix
125 Eigen::SparseMatrix<EleType> R(n_H_dofs, n_h_dofs);
126 R.reserve(P.nonZeros());
127 R = P.transpose();
128
129 // Update the maps
130 this->set_level_to_P(level, P);
131 this->set_level_to_R(level, R);
132
133 return;
134 }
